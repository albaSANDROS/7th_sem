entity d is
port (
        set: in STD_LOGIC;
        d_in: in STD_LOGIC;
        c: in STD_LOGIC;
        reset: in STD_LOGIC;
        q: out STD_LOGIC;
        not_q: out STD_LOGIC
    );
end d;

architecture Behavioral of d is
begin

    process (set, c, reset) is
    begin
        if  set = '0' then
                q <= '1';
                not_q <= '0';
            elsif reset = '0' then
                    q <= '0';
                    not_q <= '1';
                elsif rising_edge(c) then 
                    q <= d_in;
                    not_q <= not d_in;
        end if;
    end process;
end Behavioral;
